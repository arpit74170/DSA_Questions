1class Solution {
2    public int subarraySum(int[] nums, int k) {
3        int psum=0;
4        int res=0;
5        HashMap<Integer,Integer> map=new HashMap<>();
6        map.put(0,1);
7        for(int i=0;i<nums.length;i++){
8            psum=psum+nums[i];
9            if(map.containsKey(psum-k)){

12            map.put(psum,map.getOrDefault(psum,0)+1);
13        }
14        return res;
15    }
